353 /* ------------------------------------------------------------ */
354 /*** OledLineTo
355 **
356 ** Parameters:
357 ** xco - x coordinate
358 ** yco - y coordinate
359 **
360 ** Return Value:
361 ** none
362 **
363 ** Errors:
364 ** none
365 **
366 ** Description:
367 ** Draw a line from the current position to the specified
368 ** position.
369 */
370 
371 void
372 OledLineTo(int xco, int yco)
373  {
374  int err;
375  int del;
376  int lim;
377  int cpx;
378  int dxco;
379  int dyco;
380  void (*pfnMajor)();
381  void (*pfnMinor)();
382 
383  /* Clamp the point to be on the display.
384  */
385  xco = OledClampXco(xco);
386  yco = OledClampYco(yco);
387 
388  /* Determine which octant the line occupies
389  */
390  dxco = xco - xcoOledCur;
391  dyco = yco - ycoOledCur;
392  if (abs(dxco) >= abs(dyco)) {
393  /* Line is x-major
394  */
395  lim = abs(dxco);
396  del = abs(dyco);
397  if (dxco >= 0) {
398  pfnMajor = OledMoveRight;
399  }
400  else {
401  pfnMajor = OledMoveLeft;
402  }
403 
404  if (dyco >= 0) {
405  pfnMinor = OledMoveDown;
406  }
407  else {
408  pfnMinor = OledMoveUp;
409  }
410  }
411  else {
412  /* Line is y-major
413  */
414  lim = abs(dyco);
415  del = abs(dxco);
416  if (dyco >= 0) {
417  pfnMajor = OledMoveDown;
418  }
419  else {
420  pfnMajor = OledMoveUp;
421  }
422 
423  if (dxco >= 0) {
424  pfnMinor = OledMoveRight;
425  }
426  else {
427  pfnMinor = OledMoveLeft;
428  }
429  }
430 
431  /* Render the line. The algorithm is:
432  ** Write the current pixel
433  ** Move one pixel on the major axis
434  ** Add the minor axis delta to the error accumulator
435  ** if the error accumulator is greater than the major axis delta
436  ** Move one pixel in the minor axis
437  ** Subtract major axis delta from error accumulator
438  */
439  err = lim/2;
440  cpx = lim;
441  while (cpx > 0) {
442  OledDrawPixel();
443  (*pfnMajor)();
444  err += del;
445  if (err > lim) {
446  err -= lim;
447  (*pfnMinor)();
448  }
449  cpx -= 1;
450  }
451 
452  /* Update the current location variables.
453  */
454  xcoOledCur = xco;
455  ycoOledCur = yco;
456 
457 }

496 /* ------------------------------------------------------------ */
497 /*** OledFillRect
498 **
499 ** Parameters:
500 ** xco - x coordinate of other corner
501 ** yco - y coordinate of other corner
502 **
503 ** Return Value:
504 ** none
505 **
506 ** Errors:
507 ** none
508 **
509 ** Description:
510 ** Fill a rectangle bounded by the current location and
511 ** the specified location.
512 */
513 
514 void
515 OledFillRect(int xco, int yco)
516  {
517  int xcoLeft;
518  int xcoRight;
519  int ycoTop;
520  int ycoBottom;
521  int ibPat;
522  BYTE * pbCur;
523  BYTE * pbLeft;
524  int xcoCur;
525  BYTE bTmp;
526  BYTE mskPat;
527 
528  /* Clamp the point to be on the display.
529  */
530  xco = OledClampXco(xco);
531  yco = OledClampYco(yco);
532 
533  /* Set up the four sides of the rectangle.
534  */
535  if (xcoOledCur < xco) {
536  xcoLeft = xcoOledCur;
537  xcoRight = xco;
538  }
539  else {
540  xcoLeft = xco;
541  xcoRight = xcoOledCur;
542  }
543 
544  if (ycoOledCur < yco) {
545  ycoTop = ycoOledCur;
546  ycoBottom = yco;
547  }
548  else {
549  ycoTop = yco;
550  ycoBottom = ycoOledCur;
551  }
552 
553 
554  while (ycoTop <= ycoBottom) {
555  /* Compute the address of the left edge of the rectangle for this
556  ** stripe across the rectangle.
557  */
558  pbLeft = &rgbOledBmp[((ycoTop/8) * ccolOledMax) + xcoLeft];
559 
560  /* Generate a mask to preserve any low bits in the byte that aren't
561  ** part of the rectangle being filled.
562  */
563  mskPat = (1 << (ycoTop & 0x07)) - 1;
564 
565  /* Combine with a mask to preserve any upper bits in the byte that aren't
566  ** part of the rectangle being filled.
567  ** This mask will end up not preserving any bits for bytes that are in
568  ** the middle of the rectangle vertically.
569  */
570  if ((ycoTop / 8) == (ycoBottom / 8)) {
571  mskPat |= ~((1 << ((ycoBottom&0x07)+1)) - 1);
572  }
573  ibPat = xcoLeft & 0x07; //index to first pattern byte
574  xcoCur = xcoLeft;
575  pbCur = pbLeft;
576 
577  /* Loop through all of the bytes horizontally making up this stripe
578  ** of the rectangle.
579  */
580  while (xcoCur <= xcoRight) {
581  *pbCur = (*pfnDoRop)(*(pbOledPatCur+ibPat), *pbCur, ~mskPat);
582  xcoCur += 1;
583  pbCur += 1;
584  ibPat += 1;
585  if (ibPat > 7) {
586  ibPat = 0;
587  }
588  }
589 
590  /* Advance to the next horizontal stripe.
591  */
592  ycoTop = 8*((ycoTop/8)+1);
593 
594  }
595 
596 }

598 /* ------------------------------------------------------------ */
599 /*** OledGetBmp
600 **
601 ** Parameters:
602 ** dxco - width of bitmap
603 ** dyco - height of bitmap
604 ** pbBits - pointer to the bitmap bits
605 **
606 ** Return Value:
607 ** none
608 **
609 ** Errors:
610 ** none
611 **
612 ** Description:
613 ** This routine will get the bits corresponding to the
614 ** rectangle implied by the current location and the
615 ** specified width and height. The buffer specified
616 ** by pbBits must be large enough to hold the resulting
617 ** bytes.
618 */
619 
620 void
621 OledGetBmp(int dxco, int dyco, BYTE * pbBits)
622  {
623  int xcoLeft;
624  int xcoRight;
625  int ycoTop;
626  int ycoBottom;
627  BYTE * pbDspCur;
628  BYTE * pbDspLeft;
629  BYTE * pbBmpCur;
630  BYTE * pbBmpLeft;
631  int xcoCur;
632  int bnAlign;
633  BYTE mskEnd;
634  BYTE bTmp;
635 
636  /* Set up the four sides of the source rectangle.
637  */
638  xcoLeft = xcoOledCur;
639  xcoRight = xcoLeft + dxco;
640  if (xcoRight >= ccolOledMax) {
641  xcoRight = ccolOledMax - 1;
642  }
643 
644  ycoTop = ycoOledCur;
645  ycoBottom = ycoTop + dyco;
646  if (ycoBottom >= crowOledMax) {
647  ycoBottom = crowOledMax - 1;
648  }
649 
650  bnAlign = ycoTop & 0x07;
651  pbDspLeft = &rgbOledBmp[((ycoTop/8) * ccolOledMax) + xcoLeft];
652  pbBmpLeft = pbBits;
653 
654  while (ycoTop < ycoBottom) {
655 
656  if ((ycoTop / 8) == ((ycoBottom-1) / 8)) {
657  mskEnd = ((1 << (((ycoBottom-1)&0x07)+1)) - 1);
658  }
659  else {
660  mskEnd = 0xFF;
661  }
662 
663  xcoCur = xcoLeft;
664  pbDspCur = pbDspLeft;
665  pbBmpCur = pbBmpLeft;
666 
667  /* Loop through all of the bytes horizontally making up this stripe
668  ** of the rectangle.
669  */
670  if (bnAlign == 0) {
671  while (xcoCur < xcoRight) {
672  *pbBmpCur = (*pbDspCur) & mskEnd;
673  xcoCur += 1;
674  pbBmpCur += 1;
675  pbDspCur += 1;
676  }
677  }
678  else {
679  while (xcoCur < xcoRight) {
680  bTmp = *pbDspCur;
681  bTmp = *(pbDspCur+ccolOledMax);
682  *pbBmpCur = ((*pbDspCur >> bnAlign) |
683  ((*(pbDspCur+ccolOledMax)) << (8-bnAlign))) & mskEnd;
684  xcoCur += 1;
685  pbBmpCur += 1;
686  pbDspCur += 1;
687  }
688  }
689 
690  /* Advance to the next horizontal stripe.
691  */
692  ycoTop += 8;
693  pbDspLeft += ccolOledMax;
694  pbBmpLeft += dxco;
695 
696  }
697 
698 }

700 /* ------------------------------------------------------------ */
701 /*** OledPutBmp
702 **
703 ** Parameters:
704 ** dxco - width of bitmap
705 ** dyco - height of bitmap
706 ** pbBits - pointer to the bitmap bits
707 **
708 ** Return Value:
709 ** none
710 **
711 ** Errors:
712 ** none
713 **
714 ** Description:
715 ** This routine will put the specified bitmap into the display
716 ** buffer at the current location.
717 */
718 
719 void
720 OledPutBmp(int dxco, int dyco, BYTE * pbBits)
721  {
722  int xcoLeft;
723  int xcoRight;
724  int ycoTop;
725  int ycoBottom;
726  BYTE * pbDspCur;
727  BYTE * pbDspLeft;
728  BYTE * pbBmpCur;
729  BYTE * pbBmpLeft;
730  int xcoCur;
731  BYTE bDsp;
732  BYTE bBmp;
733  BYTE mskEnd;
734  BYTE mskUpper;
735  BYTE mskLower;
736  int bnAlign;
737  int fTop;
738  BYTE bTmp;
739 
740  /* Set up the four sides of the destination rectangle.
741  */
742  xcoLeft = xcoOledCur;
743  xcoRight = xcoLeft + dxco;
744  if (xcoRight >= ccolOledMax) {
745  xcoRight = ccolOledMax - 1;
746  }
747 
748  ycoTop = ycoOledCur;
749  ycoBottom = ycoTop + dyco;
750  if (ycoBottom >= crowOledMax) {
751  ycoBottom = crowOledMax - 1;
752  }
753 
754  bnAlign = ycoTop & 0x07;
755  mskUpper = (1 << bnAlign) - 1;
756  mskLower = ~mskUpper;
757  pbDspLeft = &rgbOledBmp[((ycoTop/8) * ccolOledMax) + xcoLeft];
758  pbBmpLeft = pbBits;
759  fTop = 1;
760 
761  while (ycoTop < ycoBottom) {
762  /* Combine with a mask to preserve any upper bits in the byte that aren't
763  ** part of the rectangle being filled.
764  ** This mask will end up not preserving any bits for bytes that are in
765  ** the middle of the rectangle vertically.
766  */
767  if ((ycoTop / 8) == ((ycoBottom-1) / 8)) {
768  mskEnd = ((1 << (((ycoBottom-1)&0x07)+1)) - 1);
769  }
770  else {
771  mskEnd = 0xFF;
772  }
773  if (fTop) {
774  mskEnd &= ~mskUpper;
775  }
776 
777  xcoCur = xcoLeft;
778  pbDspCur = pbDspLeft;
779  pbBmpCur = pbBmpLeft;
780 
781  /* Loop through all of the bytes horizontally making up this stripe
782  ** of the rectangle.
783  */
784  if (bnAlign == 0) {
785  while (xcoCur < xcoRight) {
786  *pbDspCur = (*pfnDoRop)(*pbBmpCur, *pbDspCur, mskEnd);
787  xcoCur += 1;
788  pbDspCur += 1;
789  pbBmpCur += 1;
790  }
791  }
792  else {
793  while (xcoCur < xcoRight) {
794  bBmp = ((*pbBmpCur) << bnAlign);
795  if (!fTop) {
796  bBmp |= ((*(pbBmpCur - dxco) >> (8-bnAlign)) & ~mskLower);
797  }
798  bBmp &= mskEnd;
799  *pbDspCur = (*pfnDoRop)(bBmp, *pbDspCur, mskEnd);
800  xcoCur += 1;
801  pbDspCur += 1;
802  pbBmpCur += 1;
803  }
804  }
805 
806  /* Advance to the next horizontal stripe.
807  */
808  ycoTop = 8*((ycoTop/8)+1);
809  pbDspLeft += ccolOledMax;
810  pbBmpLeft += dxco;
811  fTop = 0;
812 
813  }
814 
815 }

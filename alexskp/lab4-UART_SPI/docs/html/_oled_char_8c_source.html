102 /*** OledSetCursor
103 **
104 ** Parameters:
105 ** xch - horizontal character position
106 ** ych - vertical character position
107 **
108 ** Return Value:
109 ** none
110 **
111 ** Errors:
112 ** none
113 **
114 ** Description:
115 ** Set the character cursor position to the specified location.
116 ** If either the specified X or Y location is off the display, it
117 ** is clamped to be on the display.
118 */
119 
120 void
121 OledSetCursor(int xch, int ych)
122  {
123 
124  /* Clamp the specified location to the display surface
125  */
126  if (xch >= xchOledMax) {
127  xch = xchOledMax-1;
128  }
129 
130  if (ych >= ychOledMax) {
131  ych = ychOledMax-1;
132  }
133 
134  /* Save the given character location.
135  */
136  xchOledCur = xch;
137  ychOledCur = ych;
138 
139  /* Convert the character location to a frame buffer address.
140  */
141  OledMoveTo(xch*dxcoOledFontCur, ych*dycoOledFontCur);
142 
143 }

171 /* ------------------------------------------------------------ */
172 /*** OledDefUserChar
173 **
174 ** Parameters:
175 ** ch - character code to define
176 ** pbDef - definition for the character
177 **
178 ** Return Value:
179 ** none
180 **
181 ** Errors:
182 ** Returns TRUE if successful, FALSE if not
183 **
184 ** Description:
185 ** Give a definition for the glyph for the specified user
186 ** character code. User definable character codes are in
187 ** the range 0x00 - 0x1F. If the code specified by ch is
188 ** outside this range, the function returns false.
189 */
190 
191 int
192 OledDefUserChar(char ch, BYTE * pbDef)
193  {
194  BYTE * pb;
195  int ib;
196 
197  if (ch < chOledUserMax) {
198  pb = pbOledFontUser + ch * cbOledChar;
199  for (ib = 0; ib < cbOledChar; ib++) {
200  *pb++ = *pbDef++;
201  }
202  return 1;
203  }
204  else {
205  return 0;
206  }
207 
208  }

324 /*** OledDrawGlyph
325 **
326 ** Parameters:
327 ** ch - character code of character to draw
328 **
329 ** Return Value:
330 ** none
331 **
332 ** Errors:
333 ** none
334 **
335 ** Description:
336 ** Renders the specified character into the display buffer
337 ** at the current character cursor location. This does not
338 ** affect the current character cursor location or the
339 ** current drawing position in the display buffer.
340 */
341 
342 void
343 OledDrawGlyph(char ch)
344  {
345  BYTE * pbFont;
346  BYTE * pbBmp;
347  int ib;
348 
349  if ((ch & 0x80) != 0) {
350  return;
351  }
352 
353  if (ch < chOledUserMax) {
354  pbFont = pbOledFontUser + ch*cbOledChar;
355  }
356  else if ((ch & 0x80) == 0) {
357  pbFont = pbOledFontCur + (ch-chOledUserMax) * cbOledChar;
358  }
359 
360  pbBmp = pbOledCur;
361 
362  for (ib = 0; ib < dxcoOledFontCur; ib++) {
363  *pbBmp++ = *pbFont++;
364  }
365 
366 }

132  /* Initialize SPI port 2.
133  */
134  SPI2CON = 0;
135  SPI2BRG = 15; //8Mhz, with 80Mhz PB clock
136  SPI2STATbits.SPIROV = 0;
137  SPI2CONbits.CKP = 1;
138  SPI2CONbits.MSTEN = 1;
139  SPI2CONbits.ON = 1;

141  // Init value = 1 for these three bits
142  LATDSET = bitVddCtrl|bitVbatCtrl; // power off initially
143  LATDSET = bitDataCmd;
144 
145  // Set pins as outputs by clearing TRIS
146  TRISDCLR = bitVddCtrl|bitVbatCtrl;
147  TRISDCLR = bitDataCmd;

151  /* Make the RG9 pin be an output. On the Basic I/O Shield, this pin

153  */

157  // Make reset pin an output, clearing TRISG, disabling analog

178 ** Initialize the OLED display controller and turn the display on.

180 void
181 OledDspInit()
182 {
183  /* We're going to be sending commands, so clear the Data/Cmd bit
184  */
185  LATDCLR = bitDataCmd;
186  /* Start by turning VDD on and wait a while for the power to come up.
187  */
188  LATDCLR = bitVddCtrl;
189  delayMs(1);
190  /* Display off command
191  */
192  Spi2PutByte(0xAE);
193  /* Bring Reset low and then high
194  */
195  PORTGCLR = bitReset;
196  delayMs(1);
197  PORTGSET = bitReset;
198  /* Send the Set Charge Pump and Set Pre-Charge Period commands
199  */
200  Spi2PutByte(0x8D);
201  Spi2PutByte(0x14);
202  Spi2PutByte(0xD9);
203  Spi2PutByte(0xF1);
204  /* Turn on VCC and wait 100ms
205  */
206  LATDCLR = bitVbatCtrl;
207  delayMs(100);
208  /* Send the commands to invert the display. This puts the display origin
209  ** in the upper left corner.
210  *
211  */
212  Spi2PutByte(0xA1); //remap columns
213  Spi2PutByte(0xC8); //remap the rows
214  /* Send the commands to select sequential COM configuration. This makes the
215  ** display memory non-interleaved.
216  */
217  Spi2PutByte(0xDA); //set COM configuration command
218  Spi2PutByte(0x20); //sequential COM, left/right remap enabled
219  /* Send Display On command
220  */
221  Spi2PutByte(0xAF);
222 }

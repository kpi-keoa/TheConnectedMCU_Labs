68 void initWiFIREadc(void) {
69 #if defined(__32MZ2048ECG100__)
70  /* Configure AD1CON1 */
71  AD1CON1 = 0; // No AD1CON1 features are enabled including: Stop-in-Idle, early
72  // interrupt, filter delay Fractional mode and scan trigger source.
73 
74  /* Configure AD1CON2 */
75  AD1CON2 = 0; // Boost, Low-power mode off, SAMC set to min, set up the ADC Clock
76  AD1CON2bits.ADCSEL = 1; // 1 = SYSCLK, 2 REFCLK03, 3 FRC
77  AD1CON2bits.ADCDIV = 10; // DIV_20 TQ = 1/200 MHz; Tad = 10 * (TQ * 2) = 100 ns; 10 MHz ADC clock; the sweet spot
78  AD1CON2bits.SAMC = 75; // settling time is 76 TADs ((samc +1) + 4) * TAD <= 8000 (125Kbps is max, time 1/125000 = 8000 ns) => ((75 + 1) + 4) * 100 ns = 8000
79 
80  /* Configure AD1CON3 */
81  AD1CON3 = 0; // ADINSEL is not configured for this example. VREFSEL of ?0?
82  // selects AVDD and AVSS as the voltage reference.
83 
84  // AD1CON3bits.VREFSEL = 0b011; // set external VRef+/-
85 
86  /* Configure AD1GIRGENx */
87  AD1GIRQEN1 = 0; // No global interrupts are used.
88  AD1GIRQEN2 = 0;
89 
90  /* Configure AD1CSSx */
91  AD1CSS1 = 0; // No channel scanning is used.
92  AD1CSS2 = 0;
93 
94  /* Configure AD1CMPCONx */
95  AD1CMPCON1 = 0; // No digital comparators are used. Setting the AD1CMPCONx
96  AD1CMPCON2 = 0; // register to ?0? ensures that the comparator is disabled. Other
97  AD1CMPCON3 = 0; // registers are ?don?t care?.
98  AD1CMPCON4 = 0;
99  AD1CMPCON5 = 0;
100  AD1CMPCON6 = 0;
101 
102  /* Configure AD1FLTRx */
103  AD1FLTR1 = 0; // No oversampling filters are used.
104  AD1FLTR2 = 0;
105  AD1FLTR3 = 0;
106  AD1FLTR4 = 0;
107  AD1FLTR5 = 0;
108  AD1FLTR6 = 0;
109 
110  /* Set up the trigger sources */
111  AD1TRG1 = 0; // Initialize all sources to no trigger.
112  AD1TRG2 = 0;
113  AD1TRG3 = 0;
114 
115  /* Set up the CAL registers */
116  // AD1CAL1 = DEVADC1; // Copy the configuration data to the
117  // AD1CAL2 = DEVADC2; // AD1CALx special function registers.
118  // AD1CAL3 = DEVADC3;
119  // AD1CAL4 = DEVADC4;
120  // AD1CAL5 = DEVADC5;
121 
122  // comply to the errata
123  AD1CAL1 = 0xF8894530;
124  AD1CAL2 = 0x01E4AF69;
125  AD1CAL3 = 0x0FBBBBB8;
126  AD1CAL4 = 0x000004AC;
127  AD1CAL5 = 0x02000002;
128 
129  /* Turn the ADC on, start calibration */
130  AD1IMODbits.SH0MOD = 2; // put in differiential mode for self calibration
131  AD1IMODbits.SH1MOD = 2; // put in differiential mode for self calibration
132  AD1IMODbits.SH2MOD = 2; // put in differiential mode for self calibration
133  AD1IMODbits.SH3MOD = 2; // put in differiential mode for self calibration
134  AD1IMODbits.SH4MOD = 2; // put in differiential mode for self calibration
135  AD1IMODbits.SH5MOD = 2; // put in differiential mode for self calibration
136  AD1CON1bits.ADCEN = 1; // enable, start calibration
137  while (AD1CON2bits.ADCRDY == 0); // wait for calibration to complete
138  AD1IMODbits.SH0MOD = 0; // put in unipolar encoding
139  AD1IMODbits.SH1MOD = 0; // put in unipolar encoding
140  AD1IMODbits.SH2MOD = 0; // put in unipolar encoding
141  AD1IMODbits.SH3MOD = 0; // put in unipolar encoding
142  AD1IMODbits.SH4MOD = 0; // put in unipolar encoding
143  AD1IMODbits.SH5MOD = 0; // put in unipolar encoding
144 #elif defined(__32MZ2048EFG100__)
145  // initialize configuration registers
146  ADCCON1 = ADCCON2 = ADCCON3 = ADCANCON = 0;
147  // resolution 0 - 6bits, 1 - 8bits, 2 - 10bits, 3 - 12bits
148  ADCCON1bits.SELRES = 3; // shared ADC, 12 bits resolution (bits+2 TADs, 12bit resolution = 14 TAD).
149  // 0 - no trigger, 1 - clearing software trigger, 2 - not clearing software trigger, the rest see datasheet
150  ADCCON1bits.STRGSRC = 1; //Global software trigger / self clearing.
151  // 0 - internal 3.3, 1 - use external VRef+, 2 - use external VRef-
152  ADCCON3bits.VREFSEL = 0; // use internal 3.3 reference
153  // set up the TQ and TAD and S&H times
154  // TCLK: 00- pbClk3, 01 - SysClk, 10 - External Clk3, 11 - interal 8 MHz clk
155  ADCCON3bits.ADCSEL = 0b01; // TCLK clk == Sys Clock == F_CPU
156  // Global ADC TQ Clock: Global ADC prescaler 0 - 63; Divide by (CONCLKDIV*2) However, the value 0 means divide by 1
157  ADCCON3bits.CONCLKDIV = 0; // Divide by 1 == TCLK == SYSCLK == F_CPU
158  ADCCON2bits.ADCDIV = 8;
159  ADCCON2bits.SAMC = 10;
160  // the warm up count is 2^^X where X = 0 -15
161  ADCANCONbits.WKUPCLKCNT = 9; // Wakeup exponent = 2^^15 * TADx
162  /* Configure ADCIRQENx */
163  ADCCMPEN1 = ADCCMPEN2 = 0; // No interrupts are used
164  /* Configure ADCCSSx */
165  ADCCSS1 = ADCCSS2 = 0; // No scanning is used
166  /* Configure ADCCMPxCON */
167  ADCCMP1 = ADCCMP2 = ADCCMP3 = 0; // No digital comparators are used. Setting the ADCCMPxCON
168  ADCCMP4 = ADCCMP5 = ADCCMP6 = 0; // register to '0' ensures that the comparator is disabled.
169  /* Configure ADCFLTRx */
170  ADCFLTR1 = ADCFLTR2 = ADCFLTR3 = 0; // Clear all bits
171  ADCFLTR4 = ADCFLTR5 = ADCFLTR6 = 0; // Clear all bits
172  // disable all global interrupts
173  ADCGIRQEN1 = ADCGIRQEN2 = 0;
174  /* Early interrupt */
175  ADCEIEN1 = ADCEIEN2 = 0; // No early interrupt
176  // no dedicated trigger sources
177  ADCTRGMODE = 0;
178  // put everything in single ended unsigned mode
179  ADCIMCON1 = ADCIMCON2 = ADCIMCON3 = 0;
180  // triggers are all edge trigger
181  ADCTRGSNS = 0;
182  // turn on the ADC
183  ADCCON1bits.ON = 1;
184  /* Wait for voltage reference to be stable */
185  while (!ADCCON2bits.BGVRRDY); // Wait until the reference voltage is ready
186  while (ADCCON2bits.REFFLT); // Wait if there is a fault with the reference voltage
187  /* Enable clock to analog circuit */
188  ADCANCONbits.ANEN7 = 1; // Enable the clock to analog bias and digital control
189  /* Wait for ADC to be ready */
190  while (!ADCANCONbits.WKRDY7); // Wait until ADC0 is ready
191  /* Enable the ADC module */
192  ADCCON3bits.DIGEN7 = 1; // Enable shared ADC
193 #else
194 #error Unsupported processor type, please add to ADC.c.
195 #endif
196 }

200 /*** convertWiFIREadc
201  **
202  ** Parameters:
203  ** channelNumber - The PIC32 analog channel number as in the PIC32 datasheet
204  **
205  ** Return Value:
206  ** The converted value for that channel
207  **
208  ** Errors:
209  ** If return value of zero an error may have occurred
210  **
211  ** Description:
212  ** Converts the analog signal to a digital value on the
213  ** given pic32 analog channel number
214  */
215 int convertWiFIREadc(uint8_t channelNumber) {
216 #if defined(__32MZ2048ECG100__)
217  // EC uses old ADC, requires work-around to eliminate noise
218  // Samples ADC 17 times, discarding first 16 results.
219  // Uses DMA to reduce management overhead.
220 
221  int i, k = 0;
222  uint8_t vcn = channelNumber;
223  int analogValue = 0;
224  volatile unsigned td;
225 
226 #define KVA_2_PA(v) (((uint32_t) (v)) & 0x1fffffff)
227  static uint16_t __attribute__((coherent)) ovsampValue;
228 
229  // set the channel trigger for GSWTRG source triggering
230  if (channelNumber == 43 || channelNumber == 44 || channelNumber >= 50) {
231  return (0);
232  } else if (channelNumber >= 45) {
233  vcn = channelNumber - 45;
234  AD1IMOD |= 1 << ((vcn * 2) + 16); // say use the alt; set SHxALT
235  }
236 
237  AD1CON3bits.ADINSEL = vcn; // manually trigger the conversion
238  AD1CON1bits.FILTRDLY = AD1CON2bits.SAMC + 5; // strictly not needed, but set the timing anyway
239 
240  // set up for 16x oversample
241  AD1FLTR6 = 0; // clear oversampling
242  AD1FLTR6bits.OVRSAM = 1; // say 16x oversampling
243  AD1FLTR6bits.CHNLID = vcn; // the ANx channel
244 
245  // setup DMA
246  IEC4bits.DMA7IE = 0; // disable DMA channel 4 interrupts
247  IFS4bits.DMA7IF = 0; // clear existing DMA channel 4 interrupt flag
248 
249  // setup DMA channel x
250  DMACONbits.ON = 1; // make sure the DMA is ON
251  DCH7CON = 0; // clear DMA channel CON
252  DCH7ECON = 0; // clear DMA ECON
253  DCH7ECONbits.CABORT = 1; // reset the DMA channel
254  while (DCH7CONbits.CHEN == 1); // make sure DMA is not enabled
255  while (DCH7CONbits.CHBUSY == 1); // make sure DMA is not busy
256  DCH7CONbits.CHPRI = 3; // use highest priority
257  DCH7ECONbits.CHSIRQ = _ADC1_DF6_VECTOR; // say the ADC filter complete triggers the DMA
258  DCH7ECONbits.SIRQEN = 1; // enable the IRQ event for triggering
259  DCH7SSA = KVA_2_PA(&AD1FLTR6); // address of the source
260  DCH7SSIZ = 2; // source size is 2 bytes
261  DCH7CSIZ = 2; // cell size transfer
262  DCH7DSA = KVA_2_PA(&ovsampValue); // destination address
263  DCH7DSIZ = sizeof (ovsampValue); // destination size
264 
265  // must throw out first 16 samples
266  // keep the 17th. We can not access any ADC registers
267  // however we can look at the DMA to see when things complete
268  for (i = 0; i < 17; i++) {
269  do {
270 
271  DCH7ECONbits.CABORT = 1; // reset the DMA channel
272  AD1FLTR6bits.AFEN = 0; // make sure oversampling is OFF
273  while (DCH7CONbits.CHEN == 1); // wait for DMA to stop
274  while (DCH7CONbits.CHBUSY == 1); // wait for DMA not to be busy
275  DCH7INT = 0; // clear all interrupts
276  DCH7CONbits.CHEN = 1; // enable the DMA channel
277  AD1FLTR6bits.AFEN = 1; // enable oversampling
278 
279  AD1CON3bits.RQCONVRT = 1; // start conversion
280 
281  // we have noticed problems that the DMA channel is not always
282  // triggered, so after a time out value, just try again.
283  // fundamentally the problem is that AD1FLTR6bits.AFEN must
284  // be reset for each oversample conversion, disable and reenabled.
285 
286  // we know a conversion is going to take 8000ns * 16, or 128 us
287  // we don't want to check too often so DMA and ADC can work
288  // but we don't want to wait too long and hold things up
289 
290  for (k = 0; k < 20; k++) // this is more than enough time for the conversion,
291  { // really 8 should be good enough
292  for (td = 0; td < 16 * 200; td++) // Time delay
293  ;
294  if (DCH7INTbits.CHBCIF == 1) {
295  break; // DMA completed and copied the oversampled result
296  }
297  }
298  } while (DCH7INTbits.CHBCIF == 0); // if the conversion did not finish, try again
299  }
300  analogValue = ovsampValue >> 2; // 16 oversample gets you 2 extra bits
301 
302  // we are done, clean up the DMA, oversampling filter, and ADC
303  DCH7CON = 0;
304  while (DCH7CONbits.CHBUSY == 1);
305  AD1CON3bits.ADINSEL = 0;
306  AD1FLTR6 = 0;
307 
308  if (channelNumber >= 45) {
309  AD1IMOD &= ~(0b11 << ((vcn * 2) + 16)); // don't use alt
310  }
311 
312  return (analogValue);
313 #elif defined(__32MZ2048EFG100__)
314  uint32_t mask;
315 
316  if (channelNumber >= 32) // this code only supports channels 0 to 31
317  return 0;
318 
319  // Select channel to convert
320  ADCCON3bits.ADINSEL = channelNumber;
321  // Manually trigger conversion
322  ADCCON3bits.RQCNVRT = 1;
323 
324  // wait for completion of the conversion
325  mask = 0x1 << channelNumber;
326  while ((ADCDSTAT1 & mask) == 0)
327  ;
328 
329  // return the converted data
330  return ((int) ((uint32_t *) & ADCDATA0)[channelNumber]);
331 #else
332 #error Unsupported processor type, please add to ADC.c.
333 #endif
334 }

337 int ReadPotentiometerWithADC(void) {
